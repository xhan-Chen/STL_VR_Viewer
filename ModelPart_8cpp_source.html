124void ModelPart::loadSTL(QString fileName)
125{
126 QFileInfo fileInfo(fileName);
127 m_filePath = fileInfo.absoluteFilePath();
128 m_triangleCount = 0;
129 std::fill(std::begin(m_bounds), std::end(m_bounds), 0.0);
130
131 file = vtkSmartPointer<vtkSTLReader>::New();
132 file->SetFileName(fileName.toStdString().c_str());
133 file->MergingOff();
134 file->Update();
135 originalDataPort = file->GetOutputPort();
136
137 vtkPolyData* polyData = file->GetOutput();
138 if (polyData) {
139 m_triangleCount = polyData->GetNumberOfPolys();
140 polyData->GetBounds(m_bounds);
141 }
142
143 mapper = vtkSmartPointer<vtkPolyDataMapper>::New();
144 mapper->ScalarVisibilityOff();
145 mapper->StaticOff();
146
147 /* Initialize filter components */
148 shrinkFilter = vtkSmartPointer<vtkShrinkPolyData>::New();
149 shrinkFilter->SetShrinkFactor(0.8);
150
151 clipPlane = vtkSmartPointer<vtkPlane>::New();
152 clipPlane->SetOrigin(
153 (m_bounds[0] + m_bounds[1]) * 0.5,
154 (m_bounds[2] + m_bounds[3]) * 0.5,
155 (m_bounds[4] + m_bounds[5]) * 0.5);
156 clipPlane->SetNormal(1.0, 0.0, 0.0);
157
158 clipFilter = vtkSmartPointer<vtkClipPolyData>::New();
159 clipFilter->SetClipFunction(clipPlane);
160
161 actor = vtkSmartPointer<vtkActor>::New();
162 actor->SetMapper(mapper);
163
164 /* Initialize actor properties */
165 actor->GetProperty()->SetColor(m_r / 255.0, m_g / 255.0, m_b / 255.0);
166 actor->GetProperty()->SetInterpolationToPhong();
167 actor->GetProperty()->SetAmbient(0.18);
168 actor->GetProperty()->SetDiffuse(0.72);
169 actor->GetProperty()->SetSpecular(0.28);
170 actor->GetProperty()->SetSpecularPower(32.0);
171 actor->SetVisibility(this->isVisible);
172 applyTransformToActors();
173 refreshAppearance();
174
175 updatePipeline();
176}

178vtkActor* ModelPart::getNewActor() {
179 if (!file) return nullptr;
180
181 vtkSmartPointer<vtkPolyData> vrData = createGeometrySnapshot();
182 if (!vrData || vrData->GetNumberOfPoints() == 0 || vrData->GetNumberOfCells() == 0) {
183 return nullptr;
184 }
185
186 vtkSmartPointer<vtkPolyDataMapper> vrMapper = vtkSmartPointer<vtkPolyDataMapper>::New();
187 vrMapper->SetInputData(vrData);
188 vrMapper->ScalarVisibilityOff();
189 vrMapper->StaticOn();
190
191 /* Create actor instance for VR environment */
192 vtkSmartPointer<vtkActor> newVrActor = vtkSmartPointer<vtkActor>::New();
193 newVrActor->SetMapper(vrMapper);
194
195 /* Synchronize properties, visibility, and scale with existing actor */
196 if (this->actor) {
197 vtkSmartPointer<vtkProperty> vrProperty = vtkSmartPointer<vtkProperty>::New();
198 vrProperty->DeepCopy(this->actor->GetProperty());
199 newVrActor->SetProperty(vrProperty);
200 newVrActor->SetVisibility(this->actor->GetVisibility());
201 }
202
203 this->vrActor = newVrActor;
204 applyTransformToActors();
205 refreshAppearance();
206 return this->vrActor;
207}

209vtkSmartPointer<vtkPolyData> ModelPart::createGeometrySnapshot()
210{
211 vtkSmartPointer<vtkPolyData> copy = vtkSmartPointer<vtkPolyData>::New();
212 if (!file) {
213 return copy;
214 }
215
216 updatePipeline();
217
218 vtkPolyData* source = nullptr;
219 if (mapper) {
220 mapper->Update();
221 source = mapper->GetInput();
222 }
223 if (!source || source->GetNumberOfPoints() == 0 || source->GetNumberOfCells() == 0) {
224 file->Update();
225 source = file->GetOutput();
226 }
227
228 if (source) {
229 copy->DeepCopy(source);
230 }
231
232 return copy;
233}

274void ModelPart::updatePipeline()
275{
276 if (!file || !mapper) return;
277
278 mapper->StaticOff();
279
280 /* Update pipeline based on active filter combination */
281 if (!shrinkEnabled && !clipEnabled) {
282 mapper->SetInputConnection(file->GetOutputPort());
283 }
284 else if (shrinkEnabled && !clipEnabled) {
285 shrinkFilter->SetInputConnection(file->GetOutputPort());
286 mapper->SetInputConnection(shrinkFilter->GetOutputPort());
287 }
288 else if (!shrinkEnabled && clipEnabled) {
289 clipFilter->SetInputConnection(file->GetOutputPort());
290 mapper->SetInputConnection(clipFilter->GetOutputPort());
291 }
292 else {
293 /* Handle nested filter execution order */
294 if (filterOrder == 0) {
295 shrinkFilter->SetInputConnection(file->GetOutputPort());
296 clipFilter->SetInputConnection(shrinkFilter->GetOutputPort());
297 mapper->SetInputConnection(clipFilter->GetOutputPort());
298 }
299 else {
300 clipFilter->SetInputConnection(file->GetOutputPort());
301 shrinkFilter->SetInputConnection(clipFilter->GetOutputPort());
302 mapper->SetInputConnection(shrinkFilter->GetOutputPort());
303 }
304 }
305
306 mapper->Modified();
307 mapper->Update();
308 vtkPolyData* mappedData = mapper->GetInput();
309 if (mappedData && (mappedData->GetNumberOfPoints() == 0 || mappedData->GetNumberOfCells() == 0)) {
310 mapper->SetInputConnection(file->GetOutputPort());
311 mapper->Modified();
312 mapper->Update();
313 }
314
315 if (actor) {
316 actor->Modified();
317 }
318}

405void ModelPart::refreshAppearance() {
406 auto apply = [this](vtkActor* target) {
407 if (!target) return;
408
409 vtkProperty* property = target->GetProperty();
410 const double baseR = m_r / 255.0;
411 const double baseG = m_g / 255.0;
412 const double baseB = m_b / 255.0;
413
414 property->SetColor(baseR, baseG, baseB);
415 property->SetDiffuse(0.72);
416 property->SetSpecular(0.28);
417 property->SetSpecularPower(32.0);
418 property->SetAmbient(0.18);
419 property->EdgeVisibilityOff();
420 property->SetLineWidth(1.0);
421 property->SetEdgeWidth(1.0);
422 property->SetEmissiveFactor(0.0, 0.0, 0.0);
423
424 if (m_glowEnabled) {
425 const double glowR = m_glowR / 255.0;
426 const double glowG = m_glowG / 255.0;
427 const double glowB = m_glowB / 255.0;
428 property->SetAmbient(0.85);
429 property->SetDiffuse(0.45);
430 property->SetSpecular(0.75);
431 property->SetSpecularPower(90.0);
432 property->SetEmissiveFactor(glowR, glowG, glowB);
433 property->SetEdgeColor(glowR, glowG, glowB);
434 property->SetEdgeVisibility(true);
435 property->SetEdgeWidth(2.0);
436 }
437
438 if (m_highlighted || m_selectedInView) {
439 const double highlightR = m_selectedInView ? 0.15 : 1.0;
440 const double highlightG = m_selectedInView ? 0.62 : 0.78;
441 const double highlightB = m_selectedInView ? 1.0 : 0.25;
442 property->SetEdgeColor(highlightR, highlightG, highlightB);
443 property->SetEdgeVisibility(true);
444 property->SetLineWidth(m_selectedInView ? 4.0 : 3.0);
445 property->SetEdgeWidth(m_selectedInView ? 4.0 : 3.0);
446 }
447
448 target->Modified();
449 };
450
451 apply(actor);
452 apply(vrActor);
453}
